27      private static List<MemoryState> getSequenceTimeData(List<MemoryState> models,String format){        
28          LinkedHashMap<String,MemoryState> temp=new LinkedHashMap<>();
29          //将日志数据转换为统计报表数据
30          for(MemoryState item : models){
31              String key=new SimpleDateFormat(format).format(item.getRecordTime());
32              MemoryState value=temp.get(key);
33              if(value==null){
34                  value=item;
35              }else{
36                  //几次采集的内存数据，根据 已分配内存 来判断，谁大说明谁的内存使用情况最糟糕
37                  value=value.getTotalMemory()>item.getTotalMemory()?value:item;
38              }
39              
40              temp.put(key,value);
41          } 
42          List<MemoryState> list=new ArrayList<>();
43          for(MemoryState value : temp.values()){
44              list.add(value);
45          }
46          return list;
47      }

62              }

64          }
