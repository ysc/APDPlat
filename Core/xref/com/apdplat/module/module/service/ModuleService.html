180     public String toJson(Module module, ModuleFilter filter) {
181         StringBuilder json = new StringBuilder();
182         List<Module> subModules = module.getSubModules();
183 
184         if (filter != null) {
185             //是否对模块及命令进行移除交给模块控制器来做
186             filter.filter(subModules);
187         }
188 
189         if (subModules.size() > 0) {
190             json.append("[");
191             for (Module m : subModules) {
192                 json.append("{'text':'").append(m.getChinese()).append("','id':'module-").append(m.getId()).append("','iconCls':'").append(m.getEnglish()).append("'");
193                 if (filter.recursion()) {
194                     json.append(",children:").append(toJson(m, filter));
195                 }
196                 if (m.getSubModules().size() > 0) {
197                     json.append(",'leaf':false");
198                 } else {
199                     if (filter.command()) {
200                         json.append(",'leaf':false");
201                     } else {
202                         json.append(",'leaf':true");
203                     }
204                     if (filter.script()) {
205                         json.append(",listeners:{'click':function(node,event){openTab(node,event,'").append(m.getUrl()).append("')}}");
206                     }
207                 }
208                 json.append("},");
209             }
210             json = json.deleteCharAt(json.length() - 1);
211             json.append("]");
212         } else {
213             if (filter.command()) {
214                 List<Command> commands = module.getCommands();
215                 if (commands.size() > 0) {
216                     json.append("[");
217                     for (Command c : commands) {
218                         json.append("{'text':'").append(c.getChinese()).append("','id':'command-").append(c.getId()).append("','iconCls':'").append(c.getEnglish()).append("','leaf':true").append("},");
219                     }
220                     json = json.deleteCharAt(json.length() - 1);
221                     json.append("]");
222                 }
223             }
224         }
225 
226         return json.toString();
227     }

234     public void securityControl(List<Module> modules) {
235         User user = UserHolder.getCurrentLoginUser();
236         //重新装载用户，消除由于不同会话间延迟加载的问题
237         user = serviceFacade.retrieve(User.class, user.getId());
238         List<Module> userModules = user.getModule();
239         Iterator<Module> moduleIterator = modules.iterator();
240         while (moduleIterator.hasNext()) {
241             Module m = moduleIterator.next();
242             //把没有分配给用户的模块去掉
243             boolean contains = false;
244             for (Module userModule : userModules) {
245                 if (m.getId() == userModule.getId()) {
246                     contains = true;
247                     break;
248                 }
249             }
250             if (!contains) {
251                 moduleIterator.remove();
252             }
253         }
254     }

283     public static List<String> getCommandPath(Command command) {
284         List<String> result = new ArrayList<>();
285         String dependency = PropertyHolder.getProperty("command." + command.getEnglish());
286         String[] commands = null;
287         if (dependency != null && !"".equals(dependency.trim())) {
288             commands = dependency.split(",");
289         } else {
290             commands = new String[]{command.getEnglish()};
291         }
292         Module module = command.getModule();
293         //上下文不把大写字母转换为小写字母+_
294         //String modulePath = PrivilegeUtils.process(getModulePath(module.getParentModule()));
295         String modulePath = getModulePath(module.getParentModule());
296         String moduleName = PrivilegeUtils.process(module.getEnglish());
297         for (String cmd : commands) {
298             StringBuilder path = new StringBuilder();
299             path.append("/**/").append(modulePath).append(moduleName).append("!").append(cmd);
300             result.add(path.toString());
301         }
302         return result;
303     }

305     public static String getModulePath(Module module) {
306         StringBuilder str = new StringBuilder();
307         Stack<Module> stack = new Stack<>();
308         getModules(module, stack);
309         int len = stack.size();
310         for (int i = 0; i < len; i++) {
311             str.append(stack.pop().getEnglish()).append("/");
312         }
313         return str.toString();
314     }

316     private static void getModules(Module module, Stack<Module> stack) {
317         //将当前模块加入堆栈
318         stack.push(module);
319         Module parent = module.getParentModule();
320         //当还有父模块并且父模块不为根模块的时候，把父模块也加入堆栈
321         if (parent != null && !"root".equals(parent.getEnglish())) {
322             getModules(parent, stack);
323         }
324     }

327      * 比较module.xml中定义的命令和ExtJSSimpleAction中默认提供的命令
328      * 返回在在module.xml中指定而在ExtJSSimpleAction中没有默认提供的命令
329      * @param module

331      */
332     public static List<Command> getSpecialCommand(Module module) {
333         List<Command> special = new ArrayList<>();
334         List<String> commons = new ArrayList<>();
335         for (Method method : ExtJSSimpleAction.class.getMethods()) {
336             commons.add(method.getName());
337             log.info("common method: " + method.getName());
338         }
339         for (Command command : module.getCommands()) {
340             String dependency = PropertyHolder.getProperty("command." + command.getEnglish());
341             if (dependency != null && !"".equals(dependency.trim())) {
342                 for (String c : dependency.split(",")) {
343                     if (!commons.contains(c)) {
344                         log.info("被依赖的方法不存在，通常情况下，被依赖的方法应该是通用方法，应该存在");
345                         log.info("不存在通用方法: " + c);
346                     }
347                 }
348             } else {
349                 if (!commons.contains(command.getEnglish())) {
350                     special.add(command);
351                     log.info("special method:"+command.getEnglish());
352                 }
353             }
354         }
355         return special;
356     }
