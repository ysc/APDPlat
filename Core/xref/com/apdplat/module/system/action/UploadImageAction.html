93     private static void copy(File src, File dst) {
94          try {
95              InputStream in = null ;
96              OutputStream out = null ;
97              try {
98                      in = new BufferedInputStream( new FileInputStream(src), BUFFER_SIZE);
99                      out = new BufferedOutputStream( new FileOutputStream(dst), BUFFER_SIZE);
100                     byte [] buffer = new byte [BUFFER_SIZE];
101                     while (in.read(buffer) > 0 ) {
102                         out.write(buffer);
103                     }
104             } finally {
105                     if ( null != in) {
106                         in.close();
107                     }
108                     if ( null != out) {
109                         out.close();
110                     }
111             }
112         } catch (Exception e) {
113             log.error("生成验证码出错",e);
114         }
