55      public boolean common(String type,boolean backup,String date){
56          User user=UserHolder.getCurrentLoginUser();
57          String ip=UserHolder.getCurrentUserLoginIp();
58          boolean enableBackup=PropertyHolder.getBooleanProperty("monitor.backup");
59          BackupLog backupLog=null;
60          
61          if(enableBackup){
62              backupLog=new BackupLog();
63              backupLog.setOwnerUser(user);
64              backupLog.setLoginIP(ip);
65              try {
66                  backupLog.setServerIP(InetAddress.getLocalHost().getHostAddress());
67              } catch (UnknownHostException e) {
68                  log.error("记录备份日志出错",e);
69              }
70              backupLog.setAppName(SystemListener.getContextPath());
71              backupLog.setStartTime(new Date());
72              backupLog.setOperatingType(type);
73          }
74          boolean result=false;
75          try{
76              if(backup){
77                  result=backupImpl();   
78              }else{
79                  result=restoreImpl(date);   
80              }                     
81          }catch(Exception e){            
82              log.error("备份出错",e);
83          }
84          
85          if(enableBackup){
86              if(result){
87                  backupLog.setOperatingResult(BackupLogResult.SUCCESS);
88              }else{
89                  backupLog.setOperatingResult(BackupLogResult.FAIL);
90              }
91              backupLog.setEndTime(new Date());
92              backupLog.setProcessTime(backupLog.getEndTime().getTime()-backupLog.getStartTime().getTime());
93              LogQueue.addLog(backupLog);
94          }
95          return result;
96      }

119         String path=getPath();

121         File[] files=dir.listFiles();
122         for(File file : files){
123             String name=file.getName();
124             if(!name.contains("bak")) {

126             }

128             String[] temp=name.split("-");

133             String mm=temp[4];
134             String s=temp[5];
135             name=y+"-"+m+"-"+d+" "+h+":"+mm+":"+s;
