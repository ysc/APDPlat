61      private static <T> T loadClass(String keyFile,String classFile,final String className) {
62          try {
63              // DES算法要求有一个可信任的随机数源
64              SecureRandom sr = new SecureRandom();
65              byte[] rawKeyData;
66              try (InputStream fi = SecurityCheck.class.getResourceAsStream(keyFile)) {

69              // 从原始密匙数据创建一个DESKeySpec对象
70              DESKeySpec dks = new DESKeySpec(rawKeyData);
71              // 创建一个密匙工厂，然后用它把DESKeySpec对象转换成一个SecretKey对象
72              SecretKey key = SecretKeyFactory.getInstance("DES").generateSecret(dks);
73              // Cipher对象实际完成解密操作
74              Cipher cipher = Cipher.getInstance("DES");
75              // 用密匙初始化Cipher对象
76              cipher.init(Cipher.DECRYPT_MODE, key, sr);
77              byte[] encryptedData;
78              try (InputStream fi2 = SecurityCheck.class.getResourceAsStream(classFile)) {
79                  encryptedData = FileUtils.readAll(fi2);
80              }
81              // 正式执行解密操作
82              final byte decryptedData[] = cipher.doFinal(encryptedData);            
83              Object obj = new ClassLoader(){
84                  @Override
85                  public Class<?> loadClass(String name) throws ClassNotFoundException {
86                      if(className.equals(name)){
87                          return super.defineClass(name, decryptedData, 0, decryptedData.length);    
88                      }
89                      
90                      return SecurityCheck.class.getClassLoader().loadClass(name);
91                  }
92              }.loadClass(className);
93              return (T)obj;
94          } catch (IOException | InvalidKeyException | NoSuchAlgorithmException | InvalidKeySpecException | NoSuchPaddingException | IllegalBlockSizeException | BadPaddingException | ClassNotFoundException e) {
