36      @Override
37      public void prepareForDelete(Integer[] ids){
38          User loginUser=UserHolder.getCurrentLoginUser();
39          for(int id :ids){
40              UserGroup userGroup=service.retrieve(UserGroup.class, id);
41              boolean canDel=true;
42              //获取拥有等待删除的角色的所有用户
43              List<User> users=userGroup.getUsers();
44              for(User user : users){
45                  if(loginUser.getId()==user.getId()){
46                      canDel=false;
47                  }
48              }
49              if(!canDel) {
50                  continue;
51              }
52              for(User user : users){
53                  user.removeUserGroup(userGroup);
54                  service.update(user);
55              }
56          }
57      }
