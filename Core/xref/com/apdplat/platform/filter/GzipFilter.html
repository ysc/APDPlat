58      /**
59       * Performs the filtering for a request.
60       */
61      @Override
62      protected void doFilter(final HttpServletRequest request, final HttpServletResponse response,
63                              final FilterChain chain) throws Exception {
64          if (!isIncluded(request) && acceptsEncoding(request, "gzip")) {
65              // Client accepts zipped content
66              if (log.isDebugEnabled()) {
67                  log.debug(request.getRequestURL() + ". Writing with gzip compression");
68              }
69  
70              // Create a gzip stream
71              final ByteArrayOutputStream compressed = new ByteArrayOutputStream();
72              final GenericResponseWrapper wrapper;
73              try (GZIPOutputStream gzout = new GZIPOutputStream(compressed)) {
74                  wrapper = new GenericResponseWrapper(response, gzout);
75                  chain.doFilter(request, wrapper);
76                  wrapper.flush();
77              }
78  
79              //Saneness checks
80              byte[] compressedBytes = compressed.toByteArray();
81              boolean shouldGzippedBodyBeZero = ResponseUtil.shouldGzippedBodyBeZero(compressedBytes, request);
82              boolean shouldBodyBeZero = ResponseUtil.shouldBodyBeZero(request, wrapper.getStatus());
83              if (shouldGzippedBodyBeZero || shouldBodyBeZero) {
84                  compressedBytes = new byte[0];
85              }
86  
87              // Write the zipped body
88              ResponseUtil.addGzipHeader(response);
89              response.setContentLength(compressedBytes.length);
90  
91  
92              response.getOutputStream().write(compressedBytes);
93          } else {
94              // Client does not accept zipped content - don't bother zipping
95              if (log.isDebugEnabled()) {
96                  log.debug(request.getRequestURL()
97                          + ". Writing without gzip compression because the request does not accept gzip.");
98              }
99              chain.doFilter(request, response);
100         }
101     }

105     /**
106      * Checks if the request uri is an include.
107      * These cannot be gzipped.
108      */
109     private boolean isIncluded(final HttpServletRequest request) {
110         final String uri = (String) request.getAttribute("javax.servlet.include.request_uri");
111         final boolean includeRequest = !(uri == null);
112 
113         if (includeRequest && log.isDebugEnabled()) {
114             log.debug(request.getRequestURL() + " resulted in an include request. This is unusable, because" +
115                     "the response will be assembled into the overrall response. Not gzipping.");
116         }
117         return includeRequest;

380     /**
381      * Checks whether a gzipped body is actually empty and should just be zero.
382      * When the compressedBytes is {@link #EMPTY_GZIPPED_CONTENT_SIZE} it should be zero.
383      *
384      * @param compressedBytes the gzipped response body
385      * @param request         the client HTTP request
386      * @return true if the response should be 0, even if it is isn't.
387      */
388     public static boolean shouldGzippedBodyBeZero(byte[] compressedBytes, HttpServletRequest request) {
389 
390         //Check for 0 length body
391         if (compressedBytes.length == EMPTY_GZIPPED_CONTENT_SIZE) {
392             if (log.isDebugEnabled()) {
393                 log.debug(request.getRequestURL() + " resulted in an empty response.");
394             }
395             return true;
396         } else {
397             return false;
398         }

402     /**
403      * Performs a number of checks to ensure response saneness according to the rules of RFC2616:
404      * <ol>
405      * <li>If the response code is {@link javax.servlet.http.HttpServletResponse#SC_NO_CONTENT} then it is illegal for the body
406      * to contain anything. See http://www.w3.org/Protocols/rfc2616/rfc2616-sec10.html#sec10.2.5
407      * <li>If the response code is {@link javax.servlet.http.HttpServletResponse#SC_NOT_MODIFIED} then it is illegal for the body
408      * to contain anything. See http://www.w3.org/Protocols/rfc2616/rfc2616-sec10.html#sec10.3.5
409      * </ol>
410      *
411      * @param request         the client HTTP request
412      * @param responseStatus         the responseStatus
413      * @return true if the response should be 0, even if it is isn't.
414      */
415     public static boolean shouldBodyBeZero(HttpServletRequest request, int responseStatus) {
416 
417         //Check for NO_CONTENT
418         if (responseStatus == HttpServletResponse.SC_NO_CONTENT) {
419             if (log.isDebugEnabled()) {
420                 log.debug(request.getRequestURL() + " resulted in a " + HttpServletResponse.SC_NO_CONTENT
421                         + " response. Removing message body in accordance with RFC2616.");
422             }
423             return true;
424         }
425 
426         //Check for NOT_MODIFIED
427         if (responseStatus == HttpServletResponse.SC_NOT_MODIFIED) {
428             if (log.isDebugEnabled()) {
429                 log.debug(request.getRequestURL() + " resulted in a " + HttpServletResponse.SC_NOT_MODIFIED
430                         + " response. Removing message body in accordance with RFC2616.");
431             }
432             return true;
433         }
434 
435         return false;
436     }

491     /**
492      * Performs the filtering.  This method calls template method
493      * {@link #doFilter(javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse, javax.servlet.FilterChain) } which does the filtering.
494      * This method takes care of error reporting and handling.
495      * Errors are reported at {@link Log#warn(Object)} level because http tends to produce lots of errors.
496      */
497     @Override
498     public final void doFilter(final ServletRequest request, final ServletResponse response, final FilterChain chain)
499             throws ServletException {
500         final HttpServletRequest httpRequest = (HttpServletRequest) request;
501         final HttpServletResponse httpResponse = (HttpServletResponse) response;
502         try {
503             //NO_FILTER set for RequestDispatcher forwards to avoid double gzipping
504             if (filterNotDisabled(httpRequest)) {
505                 doFilter(httpRequest, httpResponse, chain);
506             } else {
507                 chain.doFilter(request, response);
508             }
509             // Flush the response
510             String implementationVendor = request.getClass().getPackage().getImplementationVendor();
511             if (implementationVendor != null && implementationVendor.equals("\"Evermind\"")) {
512                 httpResponse.getOutputStream().flush();
513             } else {
514                 httpResponse.flushBuffer();
515             }
516 
517         } catch (final Throwable throwable) {
518             log.error("error",throwable);
519         }
520     }

533     /**
534      * Checks whether a throwable, its root cause if it is a {@link ServletException}, or its cause, if it is a
535      * Chained Exception matches an entry in the exceptionsToLogDifferently list
536      *
537      * @param throwable
538      * @return true if the class name of any of the throwables is found in the exceptions to log differently
539      */
540     private boolean matches(Throwable throwable) {
541         if (exceptionsToLogDifferently == null) {
542             return false;
543         }
544         if (exceptionsToLogDifferently.indexOf(throwable.getClass().getName()) != -1) {
545             return true;
546         }
547         if (throwable instanceof ServletException) {
548             Throwable rootCause = (((ServletException) throwable).getRootCause());
549             if (exceptionsToLogDifferently.indexOf(rootCause.getClass().getName()) != -1) {
550                 return true;
551             }
552         }
553         if (throwable.getCause() != null) {
554             Throwable cause = throwable.getCause();
555             if (exceptionsToLogDifferently.indexOf(cause.getClass().getName()) != -1) {
556                 return true;
557             }
558         }
559         return false;
560     }

581     private void processInitParams(final FilterConfig config) throws ServletException {
582         String exceptions = config.getInitParameter("exceptionsToLogDifferently");
583         String level = config.getInitParameter("exceptionsToLogDifferentlyLevel");
584         String suppressStackTracesString = config.getInitParameter("suppressStackTraces");
585         suppressStackTraces = Boolean.valueOf(suppressStackTracesString).booleanValue();
586         if (log.isDebugEnabled()) {
587             log.debug("Suppression of stack traces enabled for " + this.getClass().getName());
588         }

590         if (exceptions != null) {
591             validateMandatoryParameters(exceptions, level);
592             validateLevel(level);
593             exceptionsToLogDifferentlyLevel = level;
594             exceptionsToLogDifferently = exceptions;
595             if (log.isDebugEnabled()) {

597             }
598         }

639     /**
640      * Checks if request contains the header value.
641      */
642     private boolean headerContains(final HttpServletRequest request, final String header, final String value) {
643 
644         logRequestHeaders(request);
645 
646         final Enumeration accepted = request.getHeaders(header);
647         while (accepted.hasMoreElements()) {
648             final String headerValue = (String) accepted.nextElement();
649             if (headerValue.indexOf(value) != -1) {
650                 return true;
651             }
652         }
653         return false;
654     }

656     /**
657      * Logs the request headers, if debug is enabled.
658      *
659      * @param request
660      */
661     protected void logRequestHeaders(final HttpServletRequest request) {
662         if (log.isDebugEnabled()) {
663             Map headers = new HashMap();
664             Enumeration enumeration = request.getHeaderNames();
665             StringBuilder logLine = new StringBuilder();
666             logLine.append("Request Headers");
667             while (enumeration.hasMoreElements()) {
668                 String name = (String) enumeration.nextElement();
669                 String headerValue = request.getHeader(name);
670                 headers.put(name, headerValue);
671                 logLine.append(": ").append(name).append(" -> ").append(headerValue);
672             }
673             log.debug(logLine.toString());
674         }
