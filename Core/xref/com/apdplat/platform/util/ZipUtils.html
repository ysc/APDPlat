25      /**
26       * APDPlat中的重要打包机制
27       * 将jar文件中的某个文件夹里面的内容复制到某个文件夹
28       * @param jar 包含静态资源的jar包
29       * @param subDir jar中包含待复制静态资源的文件夹名称
30       * @param loc 静态资源复制到的目标文件夹
31       * @param force 目标静态资源存在的时候是否强制覆盖
32       */
33      public static void unZip(String jar, String subDir, String loc, boolean force){
34          try {
35              File base=new File(loc);
36              if(!base.exists()){
37                  base.mkdirs();
38              }
39              
40              ZipFile zip=new ZipFile(new File(jar));
41              Enumeration<? extends ZipEntry> entrys = zip.entries();
42              while(entrys.hasMoreElements()){
43                  ZipEntry entry = entrys.nextElement();
44                  String name=entry.getName();
45                  if(!name.startsWith(subDir)){
46                      continue;
47                  }
48                  //去掉subDir
49                  name=name.replace(subDir,"").trim();
50                  if(name.length()<2){
51                      log.debug(name+" 长度 < 2");
52                      continue;
53                  }
54                  if(entry.isDirectory()){
55                      File dir=new File(base,name);
56                      if(!dir.exists()){
57                          dir.mkdirs();
58                          log.debug("创建目录");
59                      }else{
60                          log.debug("目录已经存在");
61                      }
62                      log.debug(name+" 是目录");
63                  }else{
64                      File file=new File(base,name);
65                      if(file.exists() && force){
66                          file.delete();
67                      }
68                      if(!file.exists()){
69                          InputStream in=zip.getInputStream(entry);
70                          FileUtils.copyFile(in,file);
71                          log.debug("创建文件");
72                      }else{
73                          log.debug("文件已经存在");
74                      }
75                      log.debug(name+" 不是目录");
76                  }
77              }
78          } catch (ZipException ex) {
79              log.error("文件解压失败",ex);
80          } catch (IOException ex) {
81              log.error("文件操作失败",ex);
82          }
83      }

91          FileOutputStream fos = null;
92          ZipOutputStream zos = null;
93          try {
94              fos = new FileOutputStream(zipPath);
95              zos = new ZipOutputStream(fos);
96              writeZip(new File(sourcePath), "", zos);
97          } catch (FileNotFoundException e) {
98              log.error("创建ZIP文件失败",e);
99          } finally {
100             try {
101                 if (zos != null) {
102                     zos.close();
103                 }
104             } catch (IOException e) {
105                 log.error("创建ZIP文件失败",e);
106             }
107 
108         }

111     private static void writeZip(File file, String parentPath, ZipOutputStream zos) {
112         if(file.exists()){
113             if(file.isDirectory()){//处理文件夹
114                 parentPath+=file.getName()+File.separator;
115                 File [] files=file.listFiles();
116                 for(File f:files){
117                     writeZip(f, parentPath, zos);
118                 }
119             }else{
120                 FileInputStream fis=null;
121                 try {
122                     fis=new FileInputStream(file);
123                     ZipEntry ze = new ZipEntry(parentPath + file.getName());
124                     zos.putNextEntry(ze);
125                     byte [] content=new byte[1024];
126                     int len;
127                     while((len=fis.read(content))!=-1){
128                         zos.write(content,0,len);
129                         zos.flush();
130                     }
131                     
132                 
133                 } catch (FileNotFoundException e) {
134                     log.error("创建ZIP文件失败",e);
135                 } catch (IOException e) {
136                     log.error("创建ZIP文件失败",e);
137                 }finally{
138                     try {
139                         if(fis!=null){
140                             fis.close();
141                         }
142                     }catch(IOException e){
143                         log.error("创建ZIP文件失败",e);
144                     }
145                 }
146             }
147         }
