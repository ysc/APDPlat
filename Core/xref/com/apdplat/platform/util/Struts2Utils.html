52  	public static void render(final String contentType, final String content, final String... headers) {
53  		try {
54  			//分析headers参数
55  			String encoding = ENCODING_DEFAULT;
56  			boolean noCache = NOCACHE_DEFAULT;
57  			for (String header : headers) {
58  				String headerName = StringUtils.substringBefore(header, ":");
59  				String headerValue = StringUtils.substringAfter(header, ":");
60  
61  				if (StringUtils.equalsIgnoreCase(headerName, ENCODING_PREFIX)) {
62  					encoding = headerValue;
63  				} else if (StringUtils.equalsIgnoreCase(headerName, NOCACHE_PREFIX)) {
64  					noCache = Boolean.parseBoolean(headerValue);
65  				} else {
66                                      throw new IllegalArgumentException(headerName + "不是一个合法的header类型");
67                                  }
68  			}
69  
70  			HttpServletResponse response = ServletActionContext.getResponse();
71  
72  			//设置headers参数
73  			String fullContentType = contentType + ";charset=" + encoding;
74  			response.setContentType(fullContentType);
75  			if (noCache) {
76  				response.setHeader("Pragma", "No-cache");
77  				response.setHeader("Cache-Control", "no-cache");
78  				response.setDateHeader("Expires", 0);
79  			}
80  
81  			response.getWriter().write(content);
82  			response.getWriter().flush();
83  
84  		} catch (IOException e) {
85  			log.error(e.getMessage(), e);
86  		}
87  	}
