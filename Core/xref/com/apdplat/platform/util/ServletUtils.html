69  	public static boolean checkIfNoneMatchEtag(HttpServletRequest request, HttpServletResponse response, String etag) {
70  		String headerValue = request.getHeader("If-None-Match");
71  		if (headerValue != null) {
72  			boolean conditionSatisfied = false;
73  			if (!"*".equals(headerValue)) {
74  				StringTokenizer commaTokenizer = new StringTokenizer(headerValue, ",");
75  
76  				while (!conditionSatisfied && commaTokenizer.hasMoreTokens()) {
77  					String currentToken = commaTokenizer.nextToken();
78  					if (currentToken.trim().equals(etag)) {
79  						conditionSatisfied = true;
80  					}
81  				}
82  			} else {
83  				conditionSatisfied = true;
84  			}
85  
86  			if (conditionSatisfied) {
87  				response.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
88  				response.setHeader("ETag", etag);
89  				return false;
90  			}
91  		}
92  		return true;
93  	}

104 	public static Map<String, Object> getParametersStartingWith(ServletRequest request, String prefix) {
105 		Assert.notNull(request, "Request must not be null");
106 		Enumeration paramNames = request.getParameterNames();
107 		Map<String, Object> params = new TreeMap<>();
108 		if (prefix == null) {
109 			prefix = "";
110 		}
111 		while (paramNames != null && paramNames.hasMoreElements()) {
112 			String paramName = (String) paramNames.nextElement();
113 			if ("".equals(prefix) || paramName.startsWith(prefix)) {
114 				String unprefixed = paramName.substring(prefix.length());
115 				String[] values = request.getParameterValues(paramName);
116 				if (values == null || values.length == 0) {
117 					// Do nothing, no values found at all.
118 				} else if (values.length > 1) {
119 					params.put(unprefixed, values);
120 				} else {
121 					params.put(unprefixed, values[0]);
122 				}
123 			}
124 		}
125 		return params;
126 	}
