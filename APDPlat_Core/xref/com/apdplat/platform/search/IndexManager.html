49      public void rebuidAll(){
50          if(buiding){
51              log.info("已经有任务在重建索引，当前请求无效(Rebuilding index in using, invalid request)");
52              return;
53          }
54          buiding=true;
55          final IndexLog indexLog=new IndexLog();
56          if(indexMonitor){
57              User user=UserHolder.getCurrentLoginUser();
58              indexLog.setOwnerUser(user);
59              indexLog.setLoginIP(UserHolder.getCurrentUserLoginIp());
60              try {
61                  indexLog.setServerIP(InetAddress.getLocalHost().getHostAddress());
62              } catch (UnknownHostException e) {
63                  log.error("保存索引日志出错(Error in saving index log)",e);
64              }
65              indexLog.setAppName(SystemListener.getContextPath());
66              indexLog.setStartTime(new Date());
67          }
68          new Thread(new Runnable(){
69              @Override
70              public void run(){
71                  try{
72                      log.info("开始删除索引文件(Start to delete index file)");
73                      
74                      delDir(getIndexDir());
75                      
76                      log.info("删除索引文件结束(End to delete index file)");
77                      
78                      log.info("开始建立索引文件...(Starting to create index file)");
79                      long beginTime = System.currentTimeMillis();
80                      float max=(float)Runtime.getRuntime().maxMemory()/1000000;
81                      float total=(float)Runtime.getRuntime().totalMemory()/1000000;
82                      float free=(float)Runtime.getRuntime().freeMemory()/1000000;
83                      String pre="执行之前剩余内存(Remain memory before execution):"+max+"-"+total+"+"+free+"="+(max-total+free);
84                      
85                      compassGps.index();
86                      
87                      long costTime = System.currentTimeMillis() - beginTime;
88                      max=(float)Runtime.getRuntime().maxMemory()/1000000;
89                      total=(float)Runtime.getRuntime().totalMemory()/1000000;
90                      free=(float)Runtime.getRuntime().freeMemory()/1000000;
91                      String post="执行之后剩余内存(Remain memory after execution):"+max+"-"+total+"+"+free+"="+(max-total+free);
92                      log.info("索引文件建立完毕.(Finish to build index)");
93                      log.info("花费了(this cost) " + ConvertUtils.getTimeDes(costTime));
94                      log.info(pre);
95                      log.info(post);
96                      
97                      if(indexMonitor){
98                          indexLog.setOperatingResult(IndexLogResult.SUCCESS);
99                      }
100                 }catch(CompassGpsException | IllegalStateException e){
101                     log.error("建立索引出错(Error in building index)",e);
102                     if(indexMonitor){
103                         indexLog.setOperatingResult(IndexLogResult.FAIL);
104                     }
105                 }
106                 if(indexMonitor){
107                     indexLog.setEndTime(new Date());
108                     indexLog.setProcessTime(indexLog.getEndTime().getTime()-indexLog.getStartTime().getTime());
109                     LogQueue.addLog(indexLog);
110                 }
111                 buiding=false;
112             }
113         }).start();

115     private void delDir(File file){
116         if(file.isFile()){

119             File[] files=file.listFiles();
120             if(files.length==0){

122             }else{
