115     /**
116      * Constructs actions based on a list of packages.
117      *
118      * @param configuration         The XWork configuration that the new package configs and action configs
119      *                              are added to.
120      * @param container             Xwork Container
121      * @param objectFactory         The ObjectFactory used to create the actions and such.
122      * @param redirectToSlash       A boolean parameter that controls whether or not this will create an
123      *                              action for indexes. If this is set to true, index actions are not created because
124      *                              the unknown handler will redirect from /foo to /foo/. The only action that is created
125      *                              is to the empty action in the namespace (e.g. the namespace /foo and the action "").
126      * @param defaultParentPackage  The default parent package for all the configuration.
127      */
128     @Inject
129     public APDPlatPackageBasedActionConfigBuilder(Configuration configuration, Container container, ObjectFactory objectFactory,
130                                            @Inject("struts.convention.redirect.to.slash") String redirectToSlash,
131                                            @Inject("struts.convention.default.parent.package") String defaultParentPackage) {
132 
133         // Validate that the parameters are okay
134         this.configuration = configuration;
135         this.actionNameBuilder = container.getInstance(ActionNameBuilder.class, container.getInstance(String.class, ConventionConstants.CONVENTION_ACTION_NAME_BUILDER));
136         this.resultMapBuilder = container.getInstance(ResultMapBuilder.class, container.getInstance(String.class, ConventionConstants.CONVENTION_RESULT_MAP_BUILDER));
137         this.interceptorMapBuilder = container.getInstance(InterceptorMapBuilder.class, container.getInstance(String.class, ConventionConstants.CONVENTION_INTERCEPTOR_MAP_BUILDER));
138         this.objectFactory = objectFactory;
139         this.redirectToSlash = Boolean.parseBoolean(redirectToSlash);
140 
141         if (LOG.isTraceEnabled()) {
142             LOG.trace("Setting action default parent package to [#0]", defaultParentPackage);
143         }
144 
145         this.defaultParentPackage = defaultParentPackage;
146     }

303     /**
304      * Builds the action configurations by loading all classes in the packages specified by the
305      * property <b>struts.convention.action.packages</b> and then figuring out which classes implement Action
306      * or have Action in their name. Next, if this class is in a Java package that hasn't been
307      * inspected a new PackageConfig (XWork) is created for that Java package using the Java package
308      * name. This will contain all the ActionConfigs for all the Action classes that are discovered
309      * within that Java package. Next, each class is inspected for the {@link ParentPackage}
310      * annotation which is used to control the parent package for a specific action. Lastly, the
311      * {@link ResultMapBuilder} is used to create ResultConfig instances of the action.
312      */
313     public void buildActionConfigs() {
314         //setup reload class loader based on dev settings
315         initReloadClassLoader();
316 
317         if (!disableActionScanning) {
318             if (actionPackages == null && packageLocators == null) {
319                 throw new ConfigurationException("At least a list of action packages or action package locators " +
320                         "must be given using one of the properties [struts.convention.action.packages] or " +
321                         "[struts.convention.package.locators]");
322             }
323 
324             if (LOG.isTraceEnabled()) {
325                 LOG.trace("Loading action configurations");
326                 if (actionPackages != null)
327                     LOG.trace("Actions being loaded from action packages " + Arrays.asList(actionPackages));
328                 if (packageLocators != null)
329                     LOG.trace("Actions being loaded using package locators " + Arrays.asList(packageLocators));
330                 if (excludePackages != null)
331                     LOG.trace("Excluding actions from packages " + Arrays.asList(excludePackages));
332             }
333 
334             Set<Class> classes = findActions();
335             buildConfiguration(classes);
336         }
337     }

339     protected ClassLoaderInterface getClassLoaderInterface() {
340         if (isReloadEnabled())
341             return new ClassLoaderInterfaceDelegate(this.reloadingClassLoader);
342         else {
343             /*
344             if there is a ClassLoaderInterface in the context, use it, otherwise
345             default to the default ClassLoaderInterface (a wrapper around the current
346             thread classloader)
347             using this, other plugins (like OSGi) can plugin their own classloader for a while
348             and it will be used by Convention (it cannot be a bean, as Convention is likely to be
349             called multiple times, and it needs to use the default ClassLoaderInterface during normal startup)
350             */
351             ClassLoaderInterface classLoaderInterface = null;
352             ActionContext ctx = ActionContext.getContext();
353             if (ctx != null)
354                 classLoaderInterface = (ClassLoaderInterface) ctx.get(ClassLoaderInterface.CLASS_LOADER_INTERFACE);
355 
356             return (ClassLoaderInterface) ObjectUtils.defaultIfNull(classLoaderInterface, new ClassLoaderInterfaceDelegate(getClassLoader()));
357         }
358     }

364     @SuppressWarnings("unchecked")
365     protected Set<Class> findActions() {
366         Set<Class> classes = new HashSet<Class>();
367         try {
368             if (actionPackages != null || (packageLocators != null && !disablePackageLocatorsScanning)) {
369                 ClassFinder finder = new ClassFinder(getClassLoaderInterface(), buildUrlSet().getUrls(), true, this.fileProtocols);
370 
371                 // named packages
372                 if (actionPackages != null) {
373                     for (String packageName : actionPackages) {
374                         Test<ClassFinder.ClassInfo> test = getPackageFinderTest(packageName);
375                         classes.addAll(finder.findClasses(test));
376                     }
377                 }
378 
379                 //package locators
380                 if (packageLocators != null && !disablePackageLocatorsScanning) {
381                     for (String packageLocator : packageLocators) {
382                         Test<ClassFinder.ClassInfo> test = getPackageLocatorTest(packageLocator);
383                         classes.addAll(finder.findClasses(test));
384                     }
385                 }
386             }
387         } catch (Exception ex) {
388             if (LOG.isErrorEnabled())
389                 LOG.error("Unable to scan named packages", ex);
390         }
391 
392         return classes;
393     }

395     private UrlSet buildUrlSet() throws IOException {
396         ClassLoaderInterface classLoaderInterface = getClassLoaderInterface();
397         UrlSet urlSet = new UrlSet(classLoaderInterface, this.fileProtocols);
398 
399         //excluding the urls found by the parent class loader is desired, but fails in JBoss (all urls are removed)
400         if (excludeParentClassLoader) {
401             //exclude parent of classloaders
402             ClassLoaderInterface parent = classLoaderInterface.getParent();
403             //if reload is enabled, we need to step up one level, otherwise the UrlSet will be empty
404             //this happens because the parent of the realoding class loader is the web app classloader
405             if (parent != null && isReloadEnabled())
406                 parent = parent.getParent();
407 
408             if (parent != null)
409                 urlSet = urlSet.exclude(parent);
410 
411             try {
412                 // This may fail in some sandboxes, ie GAE
413                 ClassLoader systemClassLoader = ClassLoader.getSystemClassLoader();
414                 urlSet = urlSet.exclude(new ClassLoaderInterfaceDelegate(systemClassLoader.getParent()));
415 
416             } catch (SecurityException e) {
417                 if (LOG.isWarnEnabled())
418                     LOG.warn("Could not get the system classloader due to security constraints, there may be improper urls left to scan");
419             }
420         }
421 
422         //try to find classes dirs inside war files
423         urlSet = urlSet.includeClassesUrl(classLoaderInterface);
424 
425 
426         urlSet = urlSet.excludeJavaExtDirs();
427         urlSet = urlSet.excludeJavaEndorsedDirs();
428         try {
429         	urlSet = urlSet.excludeJavaHome();
430         } catch (NullPointerException e) {
431         	// This happens in GAE since the sandbox contains no java.home directory
432             if (LOG.isWarnEnabled())
433         	    LOG.warn("Could not exclude JAVA_HOME, is this a sandbox jvm?");
434         }
435         urlSet = urlSet.excludePaths(System.getProperty("sun.boot.class.path", ""));
436         urlSet = urlSet.exclude(".*/JavaVM.framework/.*");
437 
438         if (includeJars == null) {
439             urlSet = urlSet.exclude(".*?\\.jar(!/|/)?");
440         } else {
441             Set<URL> includeUrls = new HashSet<URL>();
442             boolean[] patternUsed = new boolean[includeJars.length];
443             for (int i = 0; i < includeJars.length; i++) {
444                 try{
445                     String includeJar = includeJars[i];
446                     FileSystemResource resource=new FileSystemResource(FileUtils.getAbsolutePath(includeJar));
447                     includeUrls.add(resource.getURL());
448                     patternUsed[i] = true;
449                 }catch(Exception e){
450                     e.printStackTrace();
451                 }
452             }
453 
454             if (LOG.isWarnEnabled()) {
455                 for (int i = 0; i < patternUsed.length; i++) {
456                     if (!patternUsed[i]) {
457                         LOG.warn("The includeJars pattern [#0] did not match any jars in the classpath", includeJars[i]);
458                     }
459                 }
460             }
461             return new UrlSet(includeUrls);
462         }
463 
464         return urlSet;
465     }

467     protected Test<ClassFinder.ClassInfo> getPackageFinderTest(final String packageName) {
468         // so "my.package" does not match "my.package2.test"
469         final String strictPackageName = packageName + ".";
470         return new Test<ClassFinder.ClassInfo>() {
471             public boolean test(ClassFinder.ClassInfo classInfo) {
472                 String classPackageName = classInfo.getPackageName();
473                 boolean inPackage = classPackageName.equals(packageName) || classPackageName.startsWith(strictPackageName);
474                 boolean nameMatches = classInfo.getName().endsWith(actionSuffix);
475 
476                 try {
477                     return inPackage && (nameMatches || (checkImplementsAction && com.opensymphony.xwork2.Action.class.isAssignableFrom(classInfo.get())));
478                 } catch (ClassNotFoundException ex) {
479                     if (LOG.isErrorEnabled())
480                         LOG.error("Unable to load class [#0]", ex, classInfo.getName());
481                     return false;
482                 }
483             }
484         };
485     }

487     protected Test<ClassFinder.ClassInfo> getPackageLocatorTest(final String packageLocator) {
488         return new Test<ClassFinder.ClassInfo>() {
489             public boolean test(ClassFinder.ClassInfo classInfo) {
490                 String packageName = classInfo.getPackageName();
491                 if (packageName.length() > 0 && (packageLocatorsBasePackage == null || packageName.startsWith(packageLocatorsBasePackage))) {
492                     String[] splitted = packageName.split("\\.");
493 
494                     boolean packageMatches = StringTools.contains(splitted, packageLocator, false);
495                     boolean nameMatches = classInfo.getName().endsWith(actionSuffix);
496 
497                     try {
498                         return packageMatches && (nameMatches || (checkImplementsAction && com.opensymphony.xwork2.Action.class.isAssignableFrom(classInfo.get())));
499                     } catch (ClassNotFoundException ex) {
500                         if (LOG.isErrorEnabled())
501                             LOG.error("Unable to load class [#0]", ex, classInfo.getName());
502                         return false;
503                     }
504                 } else
505                     return false;
506             }
507         };
508     }

510     @SuppressWarnings("unchecked")
511     protected void buildConfiguration(Set<Class> classes) {
512         Map<String, PackageConfig.Builder> packageConfigs = new HashMap<String, PackageConfig.Builder>();
513 
514         for (Class<?> actionClass : classes) {
515             Actions actionsAnnotation = actionClass.getAnnotation(Actions.class);
516             Action actionAnnotation = actionClass.getAnnotation(Action.class);
517 
518             // Skip classes that can't be instantiated
519             if (cannotInstantiate(actionClass)) {
520                 if (LOG.isTraceEnabled())
521                     LOG.trace("Class [#0] did not pass the instantiation test and will be ignored", actionClass.getName());
522                 continue;
523             }
524 
525             // Tell the ObjectFactory about this class
526             try {
527                 objectFactory.getClassInstance(actionClass.getName());
528             } catch (ClassNotFoundException e) {
529                 if (LOG.isErrorEnabled())
530                     LOG.error("Object Factory was unable to load class [#0]", e, actionClass.getName());
531                 throw new StrutsException("Object Factory was unable to load class " + actionClass.getName(), e);
532             }
533 
534             // Determine the action package
535             String actionPackage = actionClass.getPackage().getName();
536             if (LOG.isDebugEnabled()) {
537                 LOG.debug("Processing class [#0] in package [#1]", actionClass.getName(), actionPackage);
538             }
539 
540             // Determine the default namespace and action name
541             List<String> namespaces = determineActionNamespace(actionClass);
542             for (String namespace : namespaces) {
543                 String defaultActionName = determineActionName(actionClass);
544                 PackageConfig.Builder defaultPackageConfig = getPackageConfig(packageConfigs, namespace,
545                         actionPackage, actionClass, null);
546 
547                 // Verify that the annotations have no errors and also determine if the default action
548                 // configuration should still be built or not.
549                 Map<String, List<Action>> map = getActionAnnotations(actionClass);
550                 Set<String> actionNames = new HashSet<String>();
551                 boolean hasDefaultMethod = ReflectionTools.containsMethod(actionClass, DEFAULT_METHOD);
552                 if (!map.containsKey(DEFAULT_METHOD)
553                         && hasDefaultMethod
554                         && actionAnnotation == null && actionsAnnotation == null
555                         && (alwaysMapExecute || map.isEmpty())) {
556                     boolean found = false;
557                     for (String method : map.keySet()) {
558                         List<Action> actions = map.get(method);
559                         for (Action action : actions) {
560 
561                             // Check if there are duplicate action names in the annotations.
562                             String actionName = action.value().equals(Action.DEFAULT_VALUE) ? defaultActionName : action.value();
563                             if (actionNames.contains(actionName)) {
564                                 throw new ConfigurationException("The action class [" + actionClass +
565                                         "] contains two methods with an action name annotation whose value " +
566                                         "is the same (they both might be empty as well).");
567                             } else {
568                                 actionNames.add(actionName);
569                             }
570 
571                             // Check this annotation is the default action
572                             if (action.value().equals(Action.DEFAULT_VALUE)) {
573                                 found = true;
574                             }
575                         }
576                     }
577 
578                     // Build the default
579                     if (!found) {
580                         createActionConfig(defaultPackageConfig, actionClass, defaultActionName, DEFAULT_METHOD, null);
581                     }
582                 }
583 
584                 // Build the actions for the annotations
585                 for (String method : map.keySet()) {
586                     List<Action> actions = map.get(method);
587                     for (Action action : actions) {
588                         PackageConfig.Builder pkgCfg = defaultPackageConfig;
589                         if (action.value().contains("/")) {
590                             pkgCfg = getPackageConfig(packageConfigs, namespace, actionPackage,
591                                     actionClass, action);
592                         }
593 
594                         createActionConfig(pkgCfg, actionClass, defaultActionName, method, action);
595                     }
596                 }
597 
598                 // some actions will not have any @Action or a default method, like the rest actions
599                 // where the action mapper is the one that finds the right method at runtime
600                 if (map.isEmpty() && mapAllMatches && actionAnnotation == null && actionsAnnotation == null) {
601                     createActionConfig(defaultPackageConfig, actionClass, defaultActionName, null, actionAnnotation);
602                 }
603 
604                 //if there are @Actions or @Action at the class level, create the mappings for them
605                 String methodName = hasDefaultMethod ? DEFAULT_METHOD : null;
606                 if (actionsAnnotation != null) {
607                     List<Action> actionAnnotations = checkActionsAnnotation(actionsAnnotation);
608                     for (Action actionAnnotation2 : actionAnnotations)
609                         createActionConfig(defaultPackageConfig, actionClass, defaultActionName, methodName, actionAnnotation2);
610                 } else if (actionAnnotation != null)
611                     createActionConfig(defaultPackageConfig, actionClass, defaultActionName, methodName, actionAnnotation);
612             }
613         }
614 
615         buildIndexActions(packageConfigs);
616 
617         // Add the new actions to the configuration
618         Set<String> packageNames = packageConfigs.keySet();
619         for (String packageName : packageNames) {
620             configuration.addPackageConfig(packageName, packageConfigs.get(packageName).build());
621         }
622     }

634     /**
635      * Determines the namespace(s) for the action based on the action class. If there is a {@link Namespace}
636      * annotation on the class (including parent classes) or on the package that the class is in, than
637      * it is used. Otherwise, the Java package name that the class is in is used in conjunction with
638      * either the <b>struts.convention.action.packages</b> or <b>struts.convention.package.locators</b>
639      * configuration values. These are used to determine which part of the Java package name should
640      * be converted into the namespace for the XWork PackageConfig.
641      *
642      * @param actionClass The action class.
643      * @return The namespace or an empty string.
644      */
645     protected List<String> determineActionNamespace(Class<?> actionClass) {
646         List<String> namespaces = new ArrayList<String>();
647 
648         // Check if there is a class or package level annotation for the namespace
649         //single namespace
650         Namespace namespaceAnnotation = AnnotationTools.findAnnotation(actionClass, Namespace.class);
651         if (namespaceAnnotation != null) {
652             if (LOG.isTraceEnabled()) {
653                 LOG.trace("Using non-default action namespace from Namespace annotation of [#0]", namespaceAnnotation.value());
654             }
655 
656             namespaces.add(namespaceAnnotation.value());
657         }
658 
659         //multiple annotations
660         Namespaces namespacesAnnotation = AnnotationTools.findAnnotation(actionClass, Namespaces.class);
661         if (namespacesAnnotation != null) {
662             if (LOG.isTraceEnabled()) {
663                 StringBuilder sb = new StringBuilder();
664                 for (Namespace namespace : namespacesAnnotation.value())
665                     sb.append(namespace.value()).append(",");
666                 sb.deleteCharAt(sb.length() - 1);
667                 LOG.trace("Using non-default action namespaces from Namespaces annotation of [#0]", sb.toString());
668             }
669 
670             for (Namespace namespace : namespacesAnnotation.value())
671                 namespaces.add(namespace.value());
672         }
673 
674         //don't use default if there are annotations
675         if (!namespaces.isEmpty())
676             return namespaces;
677 
678         String pkg = actionClass.getPackage().getName();
679         String pkgPart = null;
680         if (actionPackages != null) {
681             for (String actionPackage : actionPackages) {
682                 if (pkg.startsWith(actionPackage)) {
683                     pkgPart = actionClass.getName().substring(actionPackage.length() + 1);
684                 }
685             }
686         }
687 
688         if (pkgPart == null && packageLocators != null) {
689             for (String packageLocator : packageLocators) {
690                 int index = pkg.lastIndexOf(packageLocator);
691 
692                 // This ensures that the match is at the end, beginning or has a dot on each side of it
693                 if (index >= 0 && (index + packageLocator.length() == pkg.length() || index == 0 ||
694                         (pkg.charAt(index - 1) == '.' && pkg.charAt(index + packageLocator.length()) == '.'))) {
695                     pkgPart = actionClass.getName().substring(index + packageLocator.length() + 1);
696                 }
697             }
698         }
699 
700         if (pkgPart != null) {
701             final int indexOfDot = pkgPart.lastIndexOf('.');
702             if (indexOfDot >= 0) {
703                 String convertedNamespace = actionNameBuilder.build(pkgPart.substring(0, indexOfDot));
704                 namespaces.add("/" + convertedNamespace.replace('.', '/'));
705                 return namespaces;
706             }
707         }
708 
709         namespaces.add("");
710         return namespaces;
711     }

757      * @return a list   of Actions

759     protected List<Action> checkActionsAnnotation(Actions actionsAnnotation) {
760         Action[] actionArray = actionsAnnotation.value();
761         boolean valuelessSeen = false;
762         List<Action> actions = new ArrayList<Action>();
763         for (Action ann : actionArray) {
764             if (ann.value().equals(Action.DEFAULT_VALUE) && !valuelessSeen) {
765                 valuelessSeen = true;

790                     actionName : annotation.value();

792         }

795                 actionName, actionClass.getName());

800         }
801 
802         //build interceptors
803         List<InterceptorMapping> interceptors = interceptorMapBuilder.build(actionClass, pkgCfg, actionName, annotation);
804         actionConfig.addInterceptors(interceptors);
805 
806         //build results
807         Map<String, ResultConfig> results = resultMapBuilder.build(actionClass, annotation, actionName, pkgCfg.build());
808         actionConfig.addResultConfigs(results);
809 
810         //add params
811         if (annotation != null)
812             actionConfig.addParams(StringTools.createParameterMap(annotation.params()));
813 
814         //add exception mappings from annotation
815         if (annotation != null && annotation.exceptionMappings() != null)
816             actionConfig.addExceptionMappings(buildExceptionMappings(annotation.exceptionMappings(), actionName));
817 
818         //add exception mapping from class
819         ExceptionMappings exceptionMappings = actionClass.getAnnotation(ExceptionMappings.class);
820         if (exceptionMappings != null)
821             actionConfig.addExceptionMappings(buildExceptionMappings(exceptionMappings.value(), actionName));

824         pkgCfg.addActionConfig(actionName, actionConfig.build());

827         PackageConfig existingPkg = configuration.getPackageConfig(pkgCfg.getName());
828         if (existingPkg != null) {

859     protected PackageConfig.Builder getPackageConfig(final Map<String, PackageConfig.Builder> packageConfigs,
860                                                    String actionNamespace, final String actionPackage, final Class<?> actionClass,
861                                                    Action action) {
862         if (action != null && !action.value().equals(Action.DEFAULT_VALUE)) {
863             if (LOG.isTraceEnabled()) {
864                 LOG.trace("Using non-default action namespace from the Action annotation of [#0]", action.value());
865             }
866             String actionName = action.value();
867             actionNamespace = StringUtils.contains(actionName, "/") ? StringUtils.substringBeforeLast(actionName, "/") : StringUtils.EMPTY;
868         }
869 
870         // Next grab the parent annotation from the class
871         ParentPackage parent = AnnotationTools.findAnnotation(actionClass, ParentPackage.class);
872         String parentName = null;
873         if (parent != null) {
874             if (LOG.isTraceEnabled()) {
875                 LOG.trace("Using non-default parent package from annotation of [#0]", parent.value());
876             }
877 
878             parentName = parent.value();
879         }
880 
881         // Finally use the default
882         if (parentName == null) {
883             parentName = defaultParentPackage;
884         }
885 
886         if (parentName == null) {
887             throw new ConfigurationException("Unable to determine the parent XWork package for the action class [" +
888                     actionClass.getName() + "]");
889         }
890 
891         PackageConfig parentPkg = configuration.getPackageConfig(parentName);
892         if (parentPkg == null) {
893             throw new ConfigurationException("Unable to locate parent package [" + parentName + "]");
894         }
895 
896         // Grab based on package-namespace and if it exists, we need to ensure the existing one has
897         // the correct parent package. If not, we need to create a new package config
898         String name = actionPackage + "#" + parentPkg.getName() + "#" + actionNamespace;
899         PackageConfig.Builder pkgConfig = packageConfigs.get(name);
900         if (pkgConfig == null) {
901             pkgConfig = new PackageConfig.Builder(name).namespace(actionNamespace).addParent(parentPkg);
902             packageConfigs.put(name, pkgConfig);
903 
904             //check for @DefaultInterceptorRef in the package
905             DefaultInterceptorRef defaultInterceptorRef = AnnotationTools.findAnnotation(actionClass, DefaultInterceptorRef.class);
906             if (defaultInterceptorRef != null) {
907                 pkgConfig.defaultInterceptorRef(defaultInterceptorRef.value());
908 
909                 if (LOG.isTraceEnabled())
910                     LOG.trace("Setting [#0] as the default interceptor ref for [#1]", defaultInterceptorRef.value(), pkgConfig.getName());
911             }
912         }
913 
914         if (LOG.isTraceEnabled()) {
915             LOG.trace("Created package config named [#0] with a namespace [#1]", name, actionNamespace);
916         }
917 
918         return pkgConfig;
919     }

921     /**
922      * Determine all the index handling actions and results based on this logic:
923      *
924      * 1. Loop over all the namespaces such as /foo and see if it has an action named index
925      * 2. If an action doesn't exists in the parent namespace of the same name, create an action
926      * in the parent namespace of the same name as the namespace that points to the index
927      * action in the namespace. e.g. /foo -> /foo/index
928      * 3. Create the action in the namespace for empty string if it doesn't exist. e.g. /foo/
929      * the action is "" and the namespace is /foo
930      *
931      * @param packageConfigs Used to store the actions.
932      */
933     protected void buildIndexActions(Map<String, PackageConfig.Builder> packageConfigs) {
934         Map<String, PackageConfig.Builder> byNamespace = new HashMap<String, PackageConfig.Builder>();
935         Collection<PackageConfig.Builder> values = packageConfigs.values();
936         for (PackageConfig.Builder packageConfig : values) {
937             byNamespace.put(packageConfig.getNamespace(), packageConfig);
938         }
939 
940         // Step #1
941         Set<String> namespaces = byNamespace.keySet();
942         for (String namespace : namespaces) {
943             // First see if the namespace has an index action
944             PackageConfig.Builder pkgConfig = byNamespace.get(namespace);
945             ActionConfig indexActionConfig = pkgConfig.build().getAllActionConfigs().get("index");
946             if (indexActionConfig == null) {
947                 continue;
948             }
949 
950             // Step #2
951             if (!redirectToSlash) {
952                 int lastSlash = namespace.lastIndexOf('/');
953                 if (lastSlash >= 0) {
954                     String parentAction = namespace.substring(lastSlash + 1);
955                     String parentNamespace = namespace.substring(0, lastSlash);
956                     PackageConfig.Builder parent = byNamespace.get(parentNamespace);
957                     if (parent == null || parent.build().getAllActionConfigs().get(parentAction) == null) {
958                         if (parent == null) {
959                             parent = new PackageConfig.Builder(parentNamespace).namespace(parentNamespace).
960                                     addParents(pkgConfig.build().getParents());
961                             packageConfigs.put(parentNamespace, parent);
962                         }
963 
964                         if (parent.build().getAllActionConfigs().get(parentAction) == null) {
965                             parent.addActionConfig(parentAction, indexActionConfig);
966                         }
967                     } else if (LOG.isTraceEnabled()) {
968                         LOG.trace("The parent namespace [#0] already contains " +
969                                 "an action [#1]", parentNamespace, parentAction);
970                     }
971                 }
972             }
973 
974             // Step #3
975             if (pkgConfig.build().getAllActionConfigs().get("") == null) {
976                 if (LOG.isTraceEnabled()) {
977                     LOG.trace("Creating index ActionConfig with an action name of [] for the action " +
978                             "class [#0]", indexActionConfig.getClassName());
979                 }
980 
981                 pkgConfig.addActionConfig("", indexActionConfig);
982             }
983         }
984     }
