51  /**
52   * Default implementation of the {@link PersistenceUnitManager} interface.
53   * Used as internal default by
54   * {@link org.springframework.orm.jpa.LocalContainerEntityManagerFactoryBean}.
55   *
56   * <p>Supports standard JPA scanning for <code>persistence.xml</code> files,
57   * with configurable file locations, JDBC DataSource lookup and load-time weaving.
58   *
59   * <p>The default XML file location is <code>classpath*:META-INF/persistence.xml</code>,
60   * scanning for all matching files in the class path (as defined in the JPA specification).
61   * DataSource names are by default interpreted as JNDI names, and no load time weaving
62   * is available (which requires weaving to be turned off in the persistence provider).
63   *
64   * @author Juergen Hoeller
65   * @since 2.0
66   * @see #setPersistenceXmlLocations
67   * @see #setDataSourceLookup
68   * @see #setLoadTimeWeaver
69   * @see org.springframework.orm.jpa.LocalContainerEntityManagerFactoryBean#setPersistenceUnitManager
70   */
71  public class APDPlatPersistenceUnitManager

267 	/**
268 	 * Prepare the PersistenceUnitInfos according to the configuration
269 	 * of this manager: scanning for <code>persistence.xml</code> files,
270 	 * parsing all matching files, configurating and post-processing them.
271 	 * <p>PersistenceUnitInfos cannot be obtained before this preparation
272 	 * method has been invoked.
273 	 * @see #obtainDefaultPersistenceUnitInfo()
274 	 * @see #obtainPersistenceUnitInfo(String)
275 	 */
276 	public void preparePersistenceUnitInfos() {
277 		this.persistenceUnitInfoNames.clear();
278 		this.persistenceUnitInfos.clear();
279 		SpringPersistenceUnitInfo[] puis = readPersistenceUnitInfos();
280 		for (SpringPersistenceUnitInfo pui : puis) {
281 			if (pui.getPersistenceUnitRootUrl() == null) {
282 				pui.setPersistenceUnitRootUrl(determineDefaultPersistenceUnitRootUrl());
283 			}
284 			if (pui.getNonJtaDataSource() == null) {
285 				pui.setNonJtaDataSource(this.defaultDataSource);
286 			}
287 			if (this.loadTimeWeaver != null) {
288 				pui.init(this.loadTimeWeaver);
289 			}
290 			else {
291 				pui.init(this.resourcePatternResolver.getClassLoader());
292 			}
293 			postProcessPersistenceUnitInfo(pui);
294 			String name = pui.getPersistenceUnitName();
295 			this.persistenceUnitInfoNames.add(name);
296 			PersistenceUnitInfo puiToStore = pui;
297 			if (jpa2ApiPresent) {
298 				puiToStore = (PersistenceUnitInfo) Proxy.newProxyInstance(SmartPersistenceUnitInfo.class.getClassLoader(),
299 						new Class[] {SmartPersistenceUnitInfo.class}, new Jpa2PersistenceUnitInfoDecorator(pui));
300 			}
301 			this.persistenceUnitInfos.put(name, puiToStore);
302 		}
303 	}

317 	 * @return the persistence unit root URL to pass to the JPA PersistenceProvider
318 	 * @see #setDefaultPersistenceUnitRootLocation

320 	private URL determineDefaultPersistenceUnitRootUrl() {
321 		if (this.defaultPersistenceUnitRootLocation == null) {
322 			return null;
323 		}

333 	/**
334 	 * Return the specified PersistenceUnitInfo from this manager's cache
335 	 * of processed persistence units, keeping it in the cache (i.e. not
336 	 * 'obtaining' it for use but rather just accessing it for post-processing).
337 	 * <p>This can be used in {@link #postProcessPersistenceUnitInfo} implementations,
338 	 * detecting existing persistence units of the same name and potentially merging them.
339 	 * @param persistenceUnitName the name of the desired persistence unit
340 	 * @return the PersistenceUnitInfo in mutable form, or <code>null</code> if not available
341 	 */
342 	protected final MutablePersistenceUnitInfo getPersistenceUnitInfo(String persistenceUnitName) {
343 		PersistenceUnitInfo pui = this.persistenceUnitInfos.get(persistenceUnitName);
344 		if (pui != null && Proxy.isProxyClass(pui.getClass())) {
345 			// JPA 2.0 PersistenceUnitInfo decorator with a SpringPersistenceUnitInfo as target
346 			Jpa2PersistenceUnitInfoDecorator dec = (Jpa2PersistenceUnitInfoDecorator) Proxy.getInvocationHandler(pui);
347 			return dec.getTarget();
348 		}
349 		else {
350 			// Must be a raw JPA 1.0 SpringPersistenceUnitInfo instance
351 			return (MutablePersistenceUnitInfo) pui;
352 		}
353 	}

356 	 * Hook method allowing subclasses to customize each PersistenceUnitInfo.
357 	 * <p>Default implementation delegates to all registered PersistenceUnitPostProcessors.
358 	 * It is usually preferable to register further entity classes, jar files etc there
359 	 * rather than in a subclass of this manager, to be able to reuse the post-processors.
360 	 * @param pui the chosen PersistenceUnitInfo, as read from <code>persistence.xml</code>.
361 	 * Passed in as MutablePersistenceUnitInfo.
362 	 * @see #setPersistenceUnitPostProcessors
363 	 */
364 	protected void postProcessPersistenceUnitInfo(MutablePersistenceUnitInfo pui) {
365 		PersistenceUnitPostProcessor[] postProcessors = getPersistenceUnitPostProcessors();
366 		if (postProcessors != null) {
367 			for (PersistenceUnitPostProcessor postProcessor : postProcessors) {
368 				postProcessor.postProcessPersistenceUnitInfo(pui);
369 			}
370 		}
371 	}

429 	/**
430 	 * Decorator that exposes a JPA 2.0 compliant PersistenceUnitInfo interface for a
431 	 * JPA 1.0 based SpringPersistenceUnitInfo object, adapting the <code>getSharedCacheMode</code>
432 	 * and <code>getValidationMode</code> methods from String names to enum return values.
433 	 */
434 	private static class Jpa2PersistenceUnitInfoDecorator implements InvocationHandler {
435 
436 		private final SpringPersistenceUnitInfo target;
437 
438 		private final Class<? extends Enum> sharedCacheModeEnum;
439 
440 		private final Class<? extends Enum> validationModeEnum;
441 
442 		@SuppressWarnings("unchecked")
443 		public Jpa2PersistenceUnitInfoDecorator(SpringPersistenceUnitInfo target) {
444 			this.target = target;
445 			try {
446 				this.sharedCacheModeEnum = (Class<? extends Enum>)
447 						ClassUtils.forName("javax.persistence.SharedCacheMode", PersistenceUnitInfo.class.getClassLoader());
448 				this.validationModeEnum = (Class<? extends Enum>)
449 						ClassUtils.forName("javax.persistence.ValidationMode", PersistenceUnitInfo.class.getClassLoader());
450 			}
451 			catch (Exception ex) {
452 				throw new IllegalStateException("JPA 2.0 API enum types not present", ex);
453 			}
454 		}
455 
456 		public final SpringPersistenceUnitInfo getTarget() {
457 			return this.target;
458 		}
459 
460 		public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
461 			if (method.getName().equals("getSharedCacheMode")) {
462 				return Enum.valueOf(this.sharedCacheModeEnum, this.target.getSharedCacheModeName());
463 			}
464 			else if (method.getName().equals("getValidationMode")) {
465 				return Enum.valueOf(this.validationModeEnum, this.target.getValidationModeName());
466 			}
467 			else {
468 				return method.invoke(this.target, args);
469 			}
470 		}
471 	}
