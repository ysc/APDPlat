46       * 将多个META-INF/services/dic.xml文件进行解析，每一个文件对应一个Dic对象
47       * @return

49      public static List<Dic> getDics(){
50          List<Dic> dics=new ArrayList<>();
51          //准备数据字典DTD文件以供校验使用
52          prepareDtd();
53          try{
54              Enumeration<URL> ps = Thread.currentThread().getContextClassLoader().getResources("META-INF/services/dic.xml");
55              while(ps.hasMoreElements()) {
56                  InputStream in = null;
57                  try {
58                      URL url=ps.nextElement();
59                      log.info("找到数据字典描述文件(Find data dictionary description file)："+url.getPath());
60                      in = url.openStream();
61                      byte[] data=FileUtils.readAll(in);
62                      String xml=new String(data,"utf-8");
63                      log.info("将DTD文件替换为绝对路径(Replace DID file into absolute path)");
64                      xml=xml.replace("dic.dtd", FileUtils.getAbsolutePath(dtdFile));
65                      log.info("将数据字典文件读取到字节数组中，长度为(Load data dictionary into byte array, the length is)："+data.length);
66                      ByteArrayInputStream bin=new ByteArrayInputStream(xml.getBytes("utf-8"));    
67                      log.info("注册dic.xml文件(Register dic.xml file)");
68                      log.info("验证dic.xml文件(Verify dic.xml file )");
69                      //校验文件
70                      verifyFile(bin);
71                      // 解析数据字典
72                      parseDic(xml,dics);
73                  }catch(Exception e)
74                  {
75                      log.error("解析数据字典出错(Error in parsing the data dictionary)",e);
76                  }
77                  finally {
78                      try {
79                          in.close();
80                      } catch (IOException e) {
81                          log.error("解析数据字典出错(Error in parsing the data dictionary)",e);
82                      }
83                  }
84              }            
85          }catch(Exception e){
86              log.error("解析数据字典出错(Error in parsing the data dictionary)",e);
87          }
88          return dics;
89      }    
90      private static void parseDic(String xml, List<Dic> dics) {
91          XMLFactory factory=new XMLFactory(Dic.class,DicItem.class);
92          Dic dic=factory.unmarshal(xml);
93          //将XML表示的数据字典转变为JAVA对象表示的数据字典
94          assembleDic(dic);
95          dics.add(dic);
96      }
97      /**
98       * 采用递归的方式建立数据字典树的引用关系
99       * @param dic  数据字典树的根
100      */
101     private static void assembleDic(Dic dic){
102         Integer i=1;
103         for(DicItem c : dic.getDicItems()){
104             c.setDic(dic);
105             if(c.getOrderNum()==0){
106                 c.setOrderNum(i++);
107             }
108             if(c.getCode()==null){
109                 c.setCode(i.toString());
110             }
111         }
112         for(Dic d : dic.getSubDics()){
113             d.setParentDic(dic);
114             assembleDic(d);
115         }
116     }
117     private static void prepareDtd(){

139                 }
