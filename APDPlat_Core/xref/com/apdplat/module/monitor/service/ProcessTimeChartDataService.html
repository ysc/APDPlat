23      public static LinkedHashMap<String, Long> getProcessRate(List<ProcessTime> models) {    
24          Collections.sort(models, new Comparator(){
25  
26              @Override
27              public int compare(Object o1, Object o2) {
28                  ProcessTime p1=(ProcessTime)o1;
29                  ProcessTime p2=(ProcessTime)o2;
30                  return (int) (p1.getStartTime().getTime()-p2.getStartTime().getTime());
31              }
32          
33          });
34          LinkedHashMap<String,Long> data=new LinkedHashMap<>();
35          if(models.size()<1){
36              return data;
37          }
38          ProcessTime first=models.get(0);
39          ProcessTime latest=models.get(models.size()-1);
40          log.debug("首次请求时间："+DateTypeConverter.toDefaultDateTime(first.getStartTime()));
41          log.debug("最后请求时间："+DateTypeConverter.toDefaultDateTime(latest.getEndTime()));
42          long totalTime=latest.getEndTime().getTime()-first.getStartTime().getTime();
43          log.debug("系统总时间："+latest.getEndTime().getTime()+"-"+first.getStartTime().getTime()+"="+totalTime);
44          long processTime=0;
45          for(ProcessTime item : models){
46              log.debug("      增加请求处理时间："+item.getProcessTime());
47              processTime+=item.getProcessTime();
48          }
49          log.debug("处理请求时间："+processTime);
50          long waitTime=totalTime-processTime;
51          log.debug("系统空闲时间："+waitTime);
52          data.put("处理请求时间", processTime);
53          data.put("系统空闲时间", -waitTime);
54          
55          return data;
56      }
57      public static LinkedHashMap<String,Long> getTopData(List<ProcessTime> models, int top){

67          List<Long> list=new ArrayList<>();
68          for(Long key : keys){

70          }
71          Collections.sort(list);
72          Collections.reverse(list);
73          LinkedHashMap<String,Long> result=new LinkedHashMap<>();
74          int i=0;
75          for(Long processTime : list){
76              String newKey=DateTypeConverter.toDefaultDateTime(LinkedHashMap.get(processTime).getStartTime())+", "+LinkedHashMap.get(processTime).getResource();
77              result.put(newKey, processTime);

82          }

122     private static LinkedHashMap<String,Long> getSequenceTimeData(List<ProcessTime> models,String format){        
123         LinkedHashMap<String,ProcessTime> temp=new LinkedHashMap<>();
124         //将日志数据转换为统计报表数据
125         for(ProcessTime item : models){
126             String key=new SimpleDateFormat(format).format(item.getStartTime());
127             ProcessTime value=temp.get(key);
128             if(value==null){
129                 value=item;
130             }else{
131                 value=value.getProcessTime()>item.getProcessTime()?value:item;
132             }
133             
134             temp.put(key,value);
135         } 
136         LinkedHashMap<String,Long> LinkedHashMap=new LinkedHashMap<>();
137         for(String key : temp.keySet()){
138             LinkedHashMap.put(key+", "+temp.get(key).getResource(), temp.get(key).getProcessTime());
139         }
140         return LinkedHashMap;
141     }
