53      @PostConstruct
54      private void reGenerateModuleCss(){
55          OrderCriteria orderCriteria = new OrderCriteria();
56          orderCriteria.addOrder(new Order("id", Sequence.ASC));
57          //查出所有模块
58          List<Module> modules=serviceFacade.query(Module.class,null,null,orderCriteria).getModels();
59          StringBuilder css=new StringBuilder();
60          css.append("/* 自动生成的文件，请不要修改 */");
61          for(Module module : modules){
62              if("root".equals(module.getEnglish().trim())){
63                  //忽略根模块
64                  continue;
65              }
66              String path = ModuleService.getModulePath(module);
67              path=path.substring(0, path.length()-1);
68              css.append(".")
69                  .append(module.getEnglish())
70                  .append("{")
71                  .append("background-image: url(../images/module/")
72                  .append(path)
73                  .append(".png) !important;")
74                  .append("}");
75          }
76          FileUtils.createAndWriteFile("/platform/css/module.css", css.toString());
77          log.info("module css:"+css);

80      @PostConstruct
81      private void reGenerateCommandCss() {
82          OrderCriteria orderCriteria = new OrderCriteria();
83          orderCriteria.addOrder(new Order("id", Sequence.ASC));
84          //查出所有模块
85          List<Command> list=serviceFacade.query(Command.class,null,null,orderCriteria).getModels();
86          Set<String> commandCsses=new HashSet<>();
87          for(Command command : list){
88              //在module.xml中配置的命令可能会对于前台的几个按钮
89              String dependency = PropertyHolder.getProperty("command." + command.getEnglish());
90              String[] commands = null;
91              if (StringUtils.isNotBlank(dependency)) {
92                  commands = dependency.split(",");
93              } else {
94                  commands = new String[]{command.getEnglish()};
95              }
96              commandCsses.addAll(Arrays.asList(commands));
97          }
98          StringBuilder css=new StringBuilder();
99          css.append("/* 自动生成的文件，请不要修改 */");
100         for(String commandCss : commandCsses){
101             css.append(".")
102                 .append(commandCss)
103                 .append("{")
104                 .append("background-image: url(../images/operation/")
105                 .append(commandCss)
106                 .append(".png) !important;")
107                 .append("}");
108         }
109         FileUtils.createAndWriteFile("/platform/css/operation.css", css.toString());
110         log.info("operation css:"+css);
111     }

128     private void registeModule(Module module){
129         Page<Module> page=serviceFacade.query(Module.class);
130         if(page.getTotalRecords()==0){
131             log.info("第一次注册第一个模块: "+module.getChinese());            
132             serviceFacade.create(module);
133             registed=true;
134             //保存根模块
135             rootModule=module;
136             data.add(module);
137         }else{
138             log.info("以前已经注册过模块");
139             log.info("查找出根模块");
140             Module root=null;
141             Module existRoot=moduleService.getRootModule();
142             if(existRoot!=null){
143                 root=existRoot;
144                 log.info("找到以前导入的根模块: "+root.getChinese());
145             }else{
146                 log.info("没有找到以前导入的根模块");
147                 if(rootModule!=null){
148                     log.info("使用本次导入的根模块");
149                     root=rootModule;
150                 }
151             }
152             if(root!=null){
153                 log.info("将第一次以后的模块的根模块设置为第一次注册的根模块");
154                 for(Module subModule : module.getSubModules()){
155                     if(hasRegisteModule(subModule)){
156                         log.info("模块 "+subModule.getChinese()+" 在此前已经被注册过，此次忽略，检查其子模块");                        
157                         registeSubModule(subModule);
158                         continue;
159                     }
160                     //确保后续注册的模块的顺序在最后
161                     subModule.setOrderNum(subModule.getOrderNum()+(int)page.getTotalRecords());
162                     subModule.setParentModule(root);
163                     log.info("注册后续模块: "+subModule.getChinese());
164                     serviceFacade.create(subModule);
165                     registed=true;
166                     data.add(subModule);
167                 }
168             }else{
169                 log.info("没有找到根模块，注册失败！");
170             }
171         }
172     }
173     private void registeSubModule(Module module){        
174         log.info("模块 "+module.getChinese()+" 在此前已经被注册过，检查其命令");
175         chechCommand(module);
176         for(Module sub : module.getSubModules()){
177             if(hasRegisteModule(sub)){
178                 log.info("模块 "+sub.getChinese()+" 在此前已经被注册过，此次忽略，检查其子模块");
179                 registeSubModule(sub);
180             }else{
181                 log.info("注册后续模块: "+sub.getChinese());
182                 //重新从数据库中查询出模块，参数中的模块是从XML中解析出来的
183                 sub.setParentModule(moduleService.getModule(module.getEnglish()));
184                 serviceFacade.create(sub);
185                 registed=true;
186                 data.add(sub);
187             }
188         }

191     private void chechCommand(Module module){
192         Set<String> existCommands=new HashSet<>();
193         Module existsModule=moduleService.getModule(module.getEnglish());
194         if(existsModule!=null){
195             for(Command command : existsModule.getCommands()){
196                 existCommands.add(command.getEnglish());
197             }
198             for(Command command : module.getCommands()){
199                 if(!existCommands.contains(command.getEnglish())){
200                     command.setModule(existsModule);
201                     serviceFacade.create(command);
202                     log.info("注册新增命令: "+command.getChinese()+" ,模块："+existsModule.getChinese());
203                 }
204             }
205         }
206     }
207     private boolean hasRegisteModule(Module module){
