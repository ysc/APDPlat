48      private String getSplitString(String str, String split, int length){        
49          int len=str.length();
50          StringBuilder temp=new StringBuilder();
51          for(int i=0;i<len;i++){
52              if(i%length==0 && i>0){
53                  temp.append(split);
54              }
55              temp.append(str.charAt(i));
56          }
57          String[] attrs=temp.toString().split(split);
58          StringBuilder finalMachineCode=new StringBuilder();
59          for(String attr : attrs){
60              if(attr.length()==length){
61                  finalMachineCode.append(attr).append(split);
62              }
63          }
64          String result=finalMachineCode.toString().substring(0, finalMachineCode.toString().length()-1);
65          return result;
66      }
