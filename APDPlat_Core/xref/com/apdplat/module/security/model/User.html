251     private List<Module> assemblyModule(List<Command> commands){
252         List<Module> modules=new ArrayList<>();
253         if(commands==null) {
254             return modules;
255         }
256         
257         for(Command command : commands){
258             if(command!=null){
259                 Module module=command.getModule();
260                 if(module!=null){
261                     modules.add(module);
262                     assemblyModule(modules,module);
263                 }
264             }
265         }
266         return modules;

292         log.debug("user privilege:");
293         if(this.roles != null && !this.roles.isEmpty()) {
294             log.debug("     roles:");

296                 for (String priv : role.getAuthorities()) {
297                     log.debug(priv);
298                     grantedAuthArray.add(new GrantedAuthorityImpl(priv.toUpperCase()));
299                 }
300             }
301         }
302         if(this.userGroups != null && !this.userGroups.isEmpty()){
303             log.debug("     userGroups:");
304             for(UserGroup userGroup : this.userGroups){
305                 for(Role role : userGroup.getRoles()){
306                     for (String priv : role.getAuthorities()) {
307                         log.debug(priv);
308                         grantedAuthArray.add(new GrantedAuthorityImpl(priv.toUpperCase()));
309                     }
310                 }
311             }
312         }        
313         if(this.positions != null && !this.positions.isEmpty()) {
314             log.debug("     positions:");
315             for (Position position : this.positions) {
316                 for (String priv : position.getAuthorities()) {
317                     log.debug(priv);
318                     grantedAuthArray.add(new GrantedAuthorityImpl(priv.toUpperCase()));
319                 }
320             }
321         }
322         if(grantedAuthArray.isEmpty()){
323             return null;
324         }
325         grantedAuthArray.add(new GrantedAuthorityImpl("ROLE_MANAGER"));
